create table Client (IDNo bigserial, Fname varchar (20) not null, Mname varchar (20), Lname varchar (20) not null, Client_Type boolean not null, primary key (IDNo));

create table Donor(IDNo bigint not null, HouseNo int, Street varchar(20), Barangay varchar (20) not null, CityMun varchar (20) not null, Province varchar (20) not null, ZipCode int not null, Ethnicity varchar(20), BloodRH varchar(1) not null, BloodType varchar(2) not null, Birthday date not null, Age int, Weight numeric not null, Height numeric not null, Gallons numeric not null, amountDonated numeric not null, primary key (IDNo), foreign key (IDNo) references Client (IDNo) on update cascade on delete set null);

create table Request(RequestNo bigserial, IDNo bigint not null, Status boolean, Date date, Time time, DateNeeded date not null, RecipientName varchar(100), primary key(RequestNo), foreign key (IDNo) references Client (IDNo) on update cascade on delete set null);

create table Blood (TrackingNo bigserial, IDNo bigint not null, BloodType
varchar(2) not null, Amount numeric, Time time, Date date, WithdrawalStatus boolean, primary key (TrackingNo), foreign key (IDNo) references Client (IDNo) on update cascade on delete set null);
